Linux ile dosyaların zip ve unzip 'lenmesi 

{% highlight ruby %}
sudo apt-get install zip unzip

zip data * // dosyayı zipler
zip -r data * // alt dosyaları zipler
unzip  pics.zip
{% endhighlight %}
